<script>
  const cartContainer = document.getElementById('cartContainer');
  const totalPriceElem = document.getElementById('totalPrice');
  const buyNowBtn = document.getElementById('buyNowBtn');

  let cart = JSON.parse(localStorage.getItem('cart')) || [];

  function renderCart() {
    cartContainer.innerHTML = '';
    if (cart.length === 0) {
      cartContainer.innerHTML = '<p>Your cart is empty.</p>';
      totalPriceElem.textContent = '';
      buyNowBtn.style.display = 'none';
      return;
    }
    buyNowBtn.style.display = 'block';
    let total = 0;
    cart.forEach((item, index) => {
      const itemTotal = item.price * item.quantity;
      total += itemTotal;
      const div = document.createElement('div');
      div.classList.add('cart-item');
      div.innerHTML = `
        <img src="${item.image}" alt="${item.title}">
        <div class="details">
          <div class="title">${item.title}</div>
          <div class="options">Size: ${item.size} | Palette: ${item.palette}</div>
          <div class="price">Price: Rs. ${item.price}</div>
          <div class="quantity-control">
            <button data-index="${index}" class="decrease">-</button>
            <span>${item.quantity}</span>
            <button data-index="${index}" class="increase">+</button>
            <button data-index="${index}" class="remove-btn" title="Remove item">&times;</button>
          </div>
        </div>
      `;
      cartContainer.appendChild(div);
    });
    totalPriceElem.textContent = `Total: Rs. ${total.toLocaleString()}`;
    attachEventListeners();
  }

  function attachEventListeners() {
    document.querySelectorAll('.decrease').forEach((btn) => {
      btn.onclick = () => updateQuantity(btn.dataset.index, -1);
    });
    document.querySelectorAll('.increase').forEach((btn) => {
      btn.onclick = () => updateQuantity(btn.dataset.index, 1);
    });
    document.querySelectorAll('.remove-btn').forEach((btn) => {
      btn.onclick = () => removeItem(btn.dataset.index);
    });
  }

  function updateQuantity(index, change) {
    index = Number(index);
    cart[index].quantity += change;
    if (cart[index].quantity < 1) cart[index].quantity = 1;
    saveCart();
    renderCart();
  }

  function removeItem(index) {
    index = Number(index);
    cart.splice(index, 1);
    saveCart();
    renderCart();
  }

  function saveCart() {
    localStorage.setItem('cart', JSON.stringify(cart));
  }

  buyNowBtn.onclick = () => {
    if (cart.length === 0) {
      alert('Your cart is empty.');
      return;
    }
    alert('Proceeding to checkout...'); 
    // You can later redirect to a payment or confirmation page here.
  };

  renderCart();
</script>
